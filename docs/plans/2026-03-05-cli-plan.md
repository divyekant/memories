# cli/output.py
"""Output formatting with TTY auto-detection and JSON envelope."""
import json
import sys


def format_json_envelope(data: dict) -> str:
    return json.dumps({"ok": True, "data": data}, default=str)


def format_error_envelope(message: str, code: str = "GENERAL_ERROR") -> str:
    return json.dumps({"ok": False, "error": message, "code": code})


class OutputFormatter:
    def __init__(
        self,
        force_json: bool = False,
        force_pretty: bool = False,
        stream=None,
    ):
        if force_json:
            self.is_json = True
        elif force_pretty:
            self.is_json = False
        else:
            s = stream if stream is not None else sys.stdout
            self.is_json = not hasattr(s, "isatty") or not s.isatty()

    def success(self, data: dict) -> str:
        return format_json_envelope(data)

    def error(self, message: str, code: str = "GENERAL_ERROR") -> str:
        return format_error_envelope(message, code)

    def echo(self, data: dict, human_fn=None):
        """Print data: JSON envelope or human-formatted via human_fn."""
        if self.is_json:
            print(self.success(data))
        elif human_fn:
            human_fn(data)
        else:
            print(self.success(data))

    def echo_error(self, message: str, code: str = "GENERAL_ERROR"):
        if self.is_json:
            print(self.error(message, code), file=sys.stderr)
        else:
            import click
            click.secho(f"Error: {message}", fg="red", err=True)
```

# cli/config.py
"""Layered config resolution: flags > config file > env vars > defaults."""
import json
import os
from pathlib import Path

DEFAULT_URL = "http://localhost:8900"
DEFAULT_CONFIG_DIR = Path.home() / ".config" / "memories"


def resolve_config(
    config_dir: Path | None = None,
    flag_url: str | None = None,
    flag_api_key: str | None = None,
) -> dict:
    config_dir = config_dir or DEFAULT_CONFIG_DIR
    sources: dict[str, str] = {}

    # Layer 1: defaults
    url = DEFAULT_URL
    api_key = None
    sources["url"] = "default"
    sources["api_key"] = "default"

    # Layer 2: env vars
    if env_url := os.environ.get("MEMORIES_URL"):
        url = env_url
        sources["url"] = "env"
    if env_key := os.environ.get("MEMORIES_API_KEY"):
        api_key = env_key
        sources["api_key"] = "env"

    # Layer 3: config file
    config_file = config_dir / "config.json"
    if config_file.exists():
        try:
            file_cfg = json.loads(config_file.read_text())
            if "url" in file_cfg:
                url = file_cfg["url"]
                sources["url"] = "file"
            if "api_key" in file_cfg:
                api_key = file_cfg["api_key"]
                sources["api_key"] = "file"
        except (json.JSONDecodeError, OSError):
            pass

    # Layer 4: CLI flags (highest priority)
    if flag_url is not None:
        url = flag_url
        sources["url"] = "flag"
    if flag_api_key is not None:
        api_key = flag_api_key
        sources["api_key"] = "flag"

    return {
        "url": url,
        "api_key": api_key,
        "_sources": sources,
    }


def write_config(config_dir: Path | None = None, **kwargs) -> Path:
    config_dir = config_dir or DEFAULT_CONFIG_DIR
    config_file = config_dir / "config.json"
    config_dir.mkdir(parents=True, exist_ok=True)

    existing = {}
    if config_file.exists():
        try:
            existing = json.loads(config_file.read_text())
        except (json.JSONDecodeError, OSError):
            pass

    existing.update({k: v for k, v in kwargs.items() if v is not None})
    config_file.write_text(json.dumps(existing, indent=2) + "\n")
    return config_file
```

# cli/client.py
"""Sync HTTP client wrapping all Memories API endpoints."""
import httpx
import time


class CliConnectionError(Exception):
    """Server unreachable."""

class CliAuthError(Exception):
    """Invalid or missing API key."""

class CliNotFoundError(Exception):
    """Resource not found."""

class CliServerError(Exception):
    """Server returned 5xx."""


class MemoriesClient:
    def __init__(self, url: str, api_key: str | None = None, transport=None):
        headers = {}
        if api_key:
            headers["X-API-Key"] = api_key
        kwargs = {"base_url": url, "headers": headers, "timeout": 30.0}
        if transport:
            kwargs["transport"] = transport
        self._http = httpx.Client(**kwargs)

    def _request(self, method: str, path: str, **kwargs) -> dict:
        try:
            resp = self._http.request(method, path, **kwargs)
        except httpx.ConnectError as e:
            raise CliConnectionError(f"Cannot connect to {self._http.base_url}: {e}")
        except httpx.TimeoutException as e:
            raise CliConnectionError(f"Request timed out: {e}")

        if resp.status_code == 401:
            raise CliAuthError("Invalid or missing API key")
        if resp.status_code == 403:
            raise CliAuthError("Insufficient permissions")
        if resp.status_code == 404:
            raise CliNotFoundError(resp.json().get("detail", "Not found"))
        if resp.status_code >= 500:
            raise CliServerError(resp.json().get("detail", "Server error"))
        if resp.status_code == 422:
            detail = resp.json().get("detail", "Validation error")
            raise ValueError(f"Validation error: {detail}")

        return resp.json()

    def _get(self, path, **kwargs):
        return self._request("GET", path, **kwargs)

    def _post(self, path, **kwargs):
        return self._request("POST", path, **kwargs)

    def _patch(self, path, **kwargs):
        return self._request("PATCH", path, **kwargs)

    def _delete(self, path, **kwargs):
        return self._request("DELETE", path, **kwargs)

        return self._get("/health")

    def health_ready(self):
        return self._get("/health/ready")

    def stats(self):
        return self._get("/stats")

            body["threshold"] = threshold
        if source_prefix:
            body["source_prefix"] = source_prefix
        return self._post("/search", json=body)

    def get_memory(self, memory_id):

    def delete_memory(self, memory_id):

    def delete_batch(self, ids):

# cli/__init__.py
"""Memories CLI — full-coverage command-line interface."""
import click
from cli.config import resolve_config
from cli.client import MemoriesClient
from cli.output import OutputFormatter


class Context:
    """Shared context passed to all commands via Click context."""
    def __init__(self, client: MemoriesClient, formatter: OutputFormatter):
        self.client = client
        self.fmt = formatter


pass_ctx = click.make_pass_decorator(Context)


@click.group()
@click.option("--url", default=None, help="Memories server URL")
@click.option("--api-key", default=None, help="API key for authentication")
@click.option("--json", "force_json", is_flag=True, help="Force JSON output")
@click.option("--pretty", "force_pretty", is_flag=True, help="Force human-readable output")
@click.version_option(package_name="memories")
@click.pass_context
def app(ctx, url, api_key, force_json, force_pretty):
    """Memories — local semantic memory for AI assistants."""
    cfg = resolve_config(flag_url=url, flag_api_key=api_key)
    client = MemoriesClient(url=cfg["url"], api_key=cfg["api_key"])
    formatter = OutputFormatter(force_json=force_json, force_pretty=force_pretty)
    ctx.ensure_object(dict)
    ctx.obj = Context(client=client, formatter=formatter)


# Import and register command groups (added as tasks are completed)
from cli.commands import core  # noqa: E402, F401

# cli/commands/core.py
"""Core commands — search, add, get, list, delete, count, upsert, is-novel, folders."""
import sys
import click
from cli import app, pass_ctx
from cli.client import CliConnectionError, CliAuthError, CliNotFoundError


def handle_errors(fn):
    """Decorator to catch client errors and format them."""
    import functools

    @functools.wraps(fn)
    def wrapper(*args, **kwargs):
        ctx = args[0] if args else kwargs.get("ctx")
        try:
            return fn(*args, **kwargs)
        except CliConnectionError as e:
            ctx.fmt.echo_error(str(e), "CONNECTION_ERROR")
            raise SystemExit(3)
        except CliAuthError as e:
            ctx.fmt.echo_error(str(e), "AUTH_REQUIRED")
            raise SystemExit(4)
        except CliNotFoundError as e:
            ctx.fmt.echo_error(str(e), "NOT_FOUND")
            raise SystemExit(2)
        except Exception as e:
            ctx.fmt.echo_error(str(e), "GENERAL_ERROR")
            raise SystemExit(1)

    return wrapper


@app.command()
@click.argument("query")
@click.option("-k", "--limit", default=5, help="Number of results")
@click.option("--hybrid/--no-hybrid", default=True, help="Use hybrid search")
@click.option("--threshold", type=float, default=None, help="Minimum similarity")
@click.option("--source", default=None, help="Filter by source prefix")
@pass_ctx
@handle_errors
def search(ctx, query, limit, hybrid, threshold, source):
    """Search memories by semantic similarity."""
    data = ctx.client.search(query, k=limit, hybrid=hybrid, threshold=threshold, source_prefix=source)

    def human(d):
        results = d.get("results", [])
        if not results:
            click.echo("No results found.")
            return
        for r in results:
            score = f"{r.get('similarity', 0):.0%}"
            click.secho(f"[{r['id']}] ({score}) ", fg="cyan", nl=False)
            click.echo(r.get("source", ""), nl=False)
            click.echo()
            click.echo(f"  {r['text'][:200]}")
            click.echo()

    ctx.fmt.echo(data, human_fn=human)


@app.command()
@click.argument("text", default="-")
@click.option("--source", "-s", required=True, help="Source identifier")
@click.option("--deduplicate/--no-deduplicate", default=True, help="Skip if similar exists")
@pass_ctx
@handle_errors
def add(ctx, text, source, deduplicate):
    """Add a memory. Pass '-' or pipe stdin for text."""
    if text == "-" or (text == "-" and not sys.stdin.isatty()):
        text = sys.stdin.read().strip()
    if not text:
        raise click.UsageError("No text provided")
    data = ctx.client.add(text, source, deduplicate=deduplicate)
    ctx.fmt.echo(data, human_fn=lambda d: click.echo(f"Added memory #{d.get('id')}"))


@app.command()
@click.argument("memory_id", type=int)
@pass_ctx
@handle_errors
def get(ctx, memory_id):
    """Fetch a memory by ID."""
    data = ctx.client.get_memory(memory_id)

    def human(d):
        click.secho(f"#{d['id']}", fg="cyan", nl=False)
        click.echo(f" [{d.get('source', '')}]")
        click.echo(d.get("text", ""))
        if d.get("created_at"):
            click.secho(f"\nCreated: {d['created_at']}", dim=True)

    ctx.fmt.echo(data, human_fn=human)


@app.command("list")
@click.option("--source", default=None, help="Filter by source prefix")
@click.option("--offset", default=0, help="Starting position")
@click.option("--limit", default=20, help="Number of results (max 5000)")
@pass_ctx
@handle_errors
def list_cmd(ctx, source, offset, limit):
    """List memories with pagination."""
    data = ctx.client.list_memories(offset=offset, limit=limit, source=source)

    def human(d):
        click.echo(f"Showing {len(d.get('memories', []))} of {d.get('total', '?')} memories")
        click.echo()
        for m in d.get("memories", []):
            click.secho(f"#{m['id']}", fg="cyan", nl=False)
            click.echo(f" [{m.get('source', '')}] {m.get('text', '')[:100]}")

    ctx.fmt.echo(data, human_fn=human)


@app.command()
@click.argument("memory_id", type=int)
@pass_ctx
@handle_errors
def delete(ctx, memory_id):
    """Delete a memory by ID."""
    data = ctx.client.delete_memory(memory_id)
    ctx.fmt.echo(data, human_fn=lambda d: click.echo(f"Deleted memory #{memory_id}"))


@app.command()
@click.option("--source", default=None, help="Filter by source prefix")
@pass_ctx
@handle_errors
def count(ctx, source):
    """Count memories."""
    data = ctx.client.count(source=source)
    ctx.fmt.echo(data, human_fn=lambda d: click.echo(f"{d.get('count', 0)} memories"))


@app.command()
@click.argument("text", default="-")
@click.option("--source", "-s", required=True, help="Source identifier")
@click.option("--key", "-k", required=True, help="Stable upsert key")
@pass_ctx
@handle_errors
def upsert(ctx, text, source, key):
    """Create or update a memory by source+key."""
    if text == "-":
        text = sys.stdin.read().strip()
    if not text:
        raise click.UsageError("No text provided")
    data = ctx.client.upsert(text, source, key)
    ctx.fmt.echo(data, human_fn=lambda d: click.echo(
        f"{'Updated' if d.get('action') == 'update' else 'Created'} memory #{d.get('id')}"
    ))


@app.command("is-novel")
@click.argument("text", default="-")
@click.option("--threshold", type=float, default=0.88, help="Similarity threshold")
@pass_ctx
@handle_errors
def is_novel(ctx, text, threshold):
    """Check if text is novel (not already stored)."""
    if text == "-":
        text = sys.stdin.read().strip()
    if not text:
        raise click.UsageError("No text provided")
    data = ctx.client.is_novel(text, threshold=threshold)

    def human(d):
        if d.get("is_novel"):
            click.secho("Novel", fg="green")
        else:
            click.secho("Not novel", fg="yellow")
            ms = d.get("most_similar")
            if ms:
                click.echo(f"  Similar ({ms.get('similarity', 0):.0%}): {ms.get('text', '')[:100]}")

    ctx.fmt.echo(data, human_fn=human)


@app.command()
@pass_ctx
@handle_errors
def folders(ctx):
    """List source folders with memory counts."""
    data = ctx.client.folders()

    def human(d):
        for f in d.get("folders", []):
            click.secho(f"{f['count']:>5}", fg="cyan", nl=False)
            click.echo(f"  {f['name']}")
        click.echo(f"\nTotal: {d.get('total', 0)}")

    ctx.fmt.echo(data, human_fn=human)
